### 1. 숫자형

```python
# 정수형
>>> a = 123
>>> a
123
>>> a = 0
>>> a
0

# 실수형
>>> b = 3.3
>>> b
3.3

# type()함수를 사용해 숫자형의 타입을 알아볼 수 있음
>>> type(a)
<class 'int'>   # 정수형(Integer)
>>> type(b)
<class 'float'>   # 실수형(Floating-point)

# 사칙연산 해보기
>>> a = 3
>>> b = 4
>>> a + b   # 더하기
7
>>> a * b   # 곱하기
12
>>> a / b   # 나누기
0.75
>>> 2 ** 3   # 제곱(2의 3승)
8
>>> 15 % 4   # 15에서 4를 나눈 나머지 값을 반환
3
>>> 15 // 4   # 나눗셈 후 소수점 아랫자리를 버림
3
```


### 2. 문자열(String)

```python
# 문자열 만들기
>>> 문자열 = "문자열입니다"    # 큰따옴표로 둘러싸기 
>>> 문자열
'문자열입니다'
>>> 문자열2 = '문자열2'   # 작은따옴표로 둘러싸기 
>>> 문자열2
'문자열2'
>>> 문자열3 = """문자열3"""   # 큰따옴표 3개로 둘러싸기 
>>> 문자열3
'문자열3'
>>> 문자열4 = '''문자열4'''   # 작은따옴표 3개로 둘러싸기 
>>> 문자열4
'문자열4'
>>> 많은문자열 = """   
문자열1
문자열2
문자열3
"""
>>> 많은문자열
'\n문자열1\n문자열2\n문자열3\n'   # \n줄바꿈의미, 여러 따옴표로 문자열을 만들 수 있음

# 문자열 안에 따옴표를 포함하고 싶다면? 밖에 다른 따옴표를 붙여주기
>>> 문자열 = "'작따'"
>>> 문자열
"'작따'"
>>> 문자열 = '"큰따"'
>>> 문자열
'"큰따"'

# 문자열로 연산하기
>>> 머리 = "여러분"
>>> 꼬리 = "안녕하세요!"
>>> 머리 + 꼬리
'여러분안녕하세요!'

>>> pen = "pen"
>>> apple = "apple"
>>> pineapple = "pineapple"

>>> apple + pen
'applepen'
>>> pineapple + pen
'pineapplepen'

>>> apple * 3   # 반복문
'appleappleapple'

# 반복문 활용해보기
>>> print("//==========================================")  
//==========================================
>>> print("//" + "=" * 40)    # 짧게 코드를 작성할 수 있다.
//========================================

# 문자열 인덱싱과 슬라이싱 : 글자는 각각 인덱스를 가지고 있음!

# 이해하기
 +---+---+---+---+---+---+
 | P | y | t | h | o | n |
 +---+---+---+---+---+---+
 0   1   2   3   4   5   6
-6  -5  -4  -3  -2  -1

# 인덱싱
>>> 유튜브_채널 = "김왼손의 Khim Academy"
>>> 유튜브_채널[0]
'김'
>>> 유튜브_채널[-1]  # -는 뒤에서 부터!
'y'
>>> 유튜브_채널[5] + 유튜브_채널[6] + 유튜브_채널[7] + 유튜브_채널[8]
'Khim'

# 슬라이싱으로 더 편하게 작업하기
>>> 유튜브_채널[5:9]

>>> 유튜브_채널[:9]    # 처음부터~9까지 슬라이싱
'김왼손의 Khim'
>>> 유튜브_채널[10:]   # 10부터~마지막까지 슬라이싱

# 문자열 포맷팅

>>> print("현재 온도는 18도입니다.")

>>> type(오늘날씨)

>>> 문자열 = "가나다라마바사"
```

### 3. 리스트

```python
# 1부터 10까지의 수 중에서 홀수만 뽑고 싶다면? 예를 들어 1,3,5,7,9 라는 집합을 표현하고 싶다면? -> 리스트를 쓰자!
>>> 홀수 = [1, 3, 5, 7, 9]  # []꺽새에 들어간 것이 리스트!
>>> 홀수
[1, 3, 5, 7, 9]

# 리스트의 인덱싱과 슬라이싱
>>> 홀수[0]
1
>>> 홀수[0:2]
[1, 3]
>>> 홀수[1:] = []   # 홀수 리스트의 1번 인덱스 이후부터 지우기
>>> 홀수
[1]

# 리스트의 연산
>>> 홀수[0]
1
>>> 홀수[1]
3
>>> 홀수[0] + 홀수[2]
6
 
>>> 짝수 = [2, 4, 6]
>>> 홀수 = [1, 3, 5]
>>> 홀수 + 짝수
[1, 3, 5, 2, 4, 6]
>>> 홀수 * 5           # 홀수리스트가 5번 반복한다.
[1, 3, 5, 1, 3, 5, 1, 3, 5, 1, 3, 5, 1, 3, 5]

>>> 홀수[1] = 7        # 홀수 리스트의 1번 인덱스문자(3)를 7로 변경
>>> 홀수
[1, 7, 5]
 
# 리스트의 함수사용
# del 리스트[] : 리스트의 []인덱스 문자를 삭제
>>> 짝수
[2, 4, 6]
>>> del 짝수[1]
>>> 짝수
[2, 6]

# 리스트.append() : 리스트에 ()를 추가
>>> 짝수
[2, 6]
>>> 짝수.append(8)
>>> 짝수
[2, 6, 8]

>>> 홀수
[1]
>>> 홀수.append(3)
>>> 홀수.append(5)
>>> 홀수
[1, 3, 5]

>>> 홀수 + 짝수
[1, 3, 5, 2, 6, 8]

# 리스트.sort() : 리스트를 정렬
>>> 전체 = 홀수 + 짝수
>>> 전체
[1, 3, 5, 2, 6, 8]
>>> 전체.sort()
>>> 전체
[1, 2, 3, 5, 6, 8]
```

### 6. 집합

```python
# 집합 만들기
>>> s1 = set([1, 2, 3])    # set()의 괄호 안에 리스트 넣기
>>> s1
{1, 2, 3}

>>> s1 = set([1, 2, 3, 3, 5, 4])
>>> s1
{1, 2, 3, 4, 5}                   

>>> s2 = set("Hello")     # 문자열 넣기
>>> s2
{'H', 'l', 'e', 'o'}      
# --------------------> 중복을 허용하지 않음 / 순서가 없음

# 인덱싱으로 접근하려면? 
>>> l1 = list(s1)         # 리스트로 변환하기
>>> l1
[1, 2, 3]
>>> t1 = tuple(s1)
>>> t1
(1, 2, 3)

# 교집합 구하기 (&)
>>> s1 = set([1, 2, 3, 4, 5, 6])
>>> s2 = set([4, 5, 6, 7, 8, 9])  # 먼저 집합을 만들어줌
>>> s1 & s2
{4, 5, 6}
>>> s1.intersection(s2)   # ~.intersection() : ~와()의 교집합을 구하는 함수
{4, 5, 6}

# 합집합 구하기 (|)
>>> s1 | s2
{1, 2, 3, 4, 5, 6, 7, 8, 9}
>>> s1.union(s2)             # ~.union() : ~와()의 합집합을 구하는 함수
{1, 2, 3, 4, 5, 6, 7, 8, 9}

# 차집합 구하기 (-)
>>> s1 - s2        
{1, 2, 3}
>>> s2 - s1
{8, 9, 7}
>>> s1.difference(s2)       # ~.difference() : ~와()의 차집합을 구하는 함수
{1, 2, 3}
>>> s2.difference(s1)
{8, 9, 7}

# 집합의 함수사용
# ~.add() : 값() 1개 추가하기
>>> s1 = set([1, 2, 3])
>>> s1.add(4)
>>> s1
{1, 2, 3, 4}

# ~.update([]) : 값([]) 여러개 추가하기
>>> s1 = set([1, 2, 3])
>>> s1.update([4, 5, 6])
>>> s1
{1, 2, 3, 4, 5, 6}

# ~.remove() : 특정 값() 제거하기
>>> s1 = set([1, 2, 3])
>>> s1.remove(2)
>>> s1
{1, 3}
```



### 7. 불리언(Boolean)

```python

True

```
